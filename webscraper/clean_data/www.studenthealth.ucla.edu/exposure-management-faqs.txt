async function covidFAQ() {
    var node = document.querySelector('[data-ucla=covid-faq]');
    if(typeof(node) == undefined || node == null) {
        return;
    }
    var before = `${node.innerHTML}`;
    node.innerText = 'Loading FAQs...';
    var r = await fetch('https://www.uclahealth.org/ucla-campus-faq-feed.cfm?smajax=true');
    var data = await r.json();
    node.innerHTML = '';
    var toc = document.createElement('div');
    toc.classList.add('table-of-contents');
    node.parentNode.insertBefore(toc, node);
    console.log(node.parentNode.parentNode);
    var activeCategory = '';
    for(var i=0; i<data.length; i++) {
        var item = data[i];
        if(activeCategory != item.category) {
            var h2 = document.createElement('h2');
            h2.innerHTML = item.category;
            activeCategory = `${item.category}`;
            node.appendChild(h2);
        }
        var h3 = document.createElement('h3');
        h3.innerHTML = item.question;

        var div = document.createElement('div');
        div.innerHTML = item.answer;

        for(var ul of div.querySelectorAll('ul')) {
            ul.classList.add('bullet');
        }
        for(var a of div.querySelectorAll('a')) {
            if(!a.getAttribute('href').match(/http/)) {
                a.setAttribute('href', `https:///www.uclahealth.org/${a.getAttribute('href')}`);
            }
        }

        node.appendChild(h3);
        node.appendChild(div);
    }

var tocs = document.querySelectorAll('.table-of-contents');

function generateTableOfContents(toc) {
    var parent = document.querySelector('.page-content');
    toc.innerHTML = '';
    var currentLevel = 2;

        2: document.createElement('ul'),

    }

    console.log(items);
    for(var i=0; i < items.length; i++) {
        var item = items[i];
        console.log(item);
        var key = item.textContent.replace(/ /g, '_').replace(/[^a-zA-Z0-9_]/g, '').toLowerCase();
        item.id = key;
        var li = document.createElement('li');        
        var a = document.createElement('a');
        a.href = '#' + key;
        a.innerText = item.innerText;
        li.appendChild(a);
        var level = parseInt(item.nodeName.replace('H', ''));

    }

  <div class="body-content paragraph--section" data-section-title="Positive COVID-19 Test Result">
          <h2>
Positive COVID-19 Test Result
</h2>
      
<h3><strong>I just tested positive for COVID. What do I do?</strong></h3>

<p>If you tested positive, isolate immediately and report your positive test to the Exposure Management Team at <a href="mailto:exposuremanagementteam@ashe.ucla.edu"><strong>exposuremanagementteam@ashe.ucla.edu</strong></a><strong> or (310) 206-6217. </strong>While waiting to be contacted, please notify any close contacts of your positive test.</p>

<h3><strong>What is the isolation timeline for a positive COVID-19 result? When can I come to campus or leave isolation?</strong></h3>

<p>The standard isolation period for those who test positive for COVID-19 is 5 days if you have not had a fever for at least 24 hours without the use of fever-reducing medication, <strong>AND</strong> your symptoms are mild and improving, or have never developed. See further isolation guidance listed in the&nbsp;<a href="https://ucla.app.box.com/s/ytuwjy4j3ntmvfuexfga6jxmwseu6ghd">UCLA COVID-19 Isolation and Quarantine Protocol</a>.</p>

<h3><strong>I recently did a rapid test at home and it was positive. What should I do?</strong></h3>

<p>You need to isolate immediately.&nbsp;<a href="https://www4.studenthealth.ucla.edu/contactus/contactus.aspx?id=covid">Alert the Exposure Management Team</a>. The team&nbsp;will contact you for case investigation as soon as possible.</p>

<h3><strong>How do I self-request a diagnostic test via The Ashe Center Patient Portal if I have symptoms?</strong></h3>

<p>Log into The Ashe Center Patient Portal&nbsp;&gt; Appointments/Online Services &gt; Schedule an Appointment &gt; COVID-19 testing.&nbsp;An Ashe Center nurse will contact you and provide testing instructions within one business day. Results from diagnostic testing at Ashe are typically available the next morning.</p>

      </div>

  <div class="body-content paragraph--section" data-section-title="UCLA COVID-19 Vaccination Policy">
          <h2>
UCLA COVID-19 Vaccination Policy
</h2>
      
<h3><strong>Am I required to get the COVID-19 booster?</strong></h3>

<p>No, the COVID-19 booster is no longer required per the <a href="https://ucnet.universityofcalifornia.edu/news/2023/05/uc-updating-covid-19-vaccination-policy-to-systemwide-opt-out-program.html">UC Office of the President</a>.</p>

<h3><strong>What is the new COVID-19 vaccination policy?</strong></h3>

<p>Under the revised policy that will be circulated for review, the University will require students and employees to <strong>either receive or affirmatively decline</strong> COVID-19 vaccination. It is expected that the revised policy will require covered individuals to be vaccinated consistent with applicable public health recommendations or opt out. Neither the Food and Drug Administration (FDA) nor the Centers for Disease Control and Prevention (CDC) has yet made a recommendation on the frequency of vaccination.</p>

<h3><strong>What inspired the changes to the COVID-19 vaccination policy?</strong></h3>

<p>The decision to update UC’s policy reflects the welcome fact that UC has entered a very different phase in our response, with&nbsp;<a href="https://www.cdc.gov/coronavirus/2019-ncov/vaccines/stay-up-to-date.html">more than 88 million COVID-19 vaccines having been administered in California</a>.&nbsp;The CDC&nbsp;<a href="https://www.cdc.gov/coronavirus/2019-ncov/vaccines/stay-up-to-date.html">continues to recommend that everyone 6 months of age and older be up to date with COVID vaccines</a>.</p>

<h3><strong>I have questions or concerns about this updated COVID-19 vaccination policy. Where should I send them?</strong></h3>

<p>If you have questions or concerns about the proposed policy change, please direct them to:&nbsp;<a href="mailto:vaccinepolicycomments@ucop.edu">vaccinepolicycomments@ucop.edu.</a></p>

<h3><strong>Where can I learn more about these changes to the COVID-19 vaccination policy?</strong></h3>

<p>Visit the <a href="https://covid-19.ucla.edu/">UCLA COVID-19 and Vaccine Resources </a>hub for the latest news, and updates on UCLA's COVID-19 vaccination policy.&nbsp;</p>

      </div>

  <div class="body-content paragraph--section" data-section-title="UCLA Campus Housing and Isolation">
          <h2>
UCLA Campus Housing and Isolation
</h2>
      
<h3><strong>I live in the on-campus dorms and I tested positive. Do I need to move into isolation housing? </strong></h3>

<p>UCLA has a very limited number of isolation spaces available to students living on-campus. Students who test positive will have the option to utilize one of these rooms as space permits. More information on COVID-19 protocols for UCLA Housing residents is <a href="https://portal.housing.ucla.edu/covid-19-protocols">available through the UCLA Housing portal.</a></p>
