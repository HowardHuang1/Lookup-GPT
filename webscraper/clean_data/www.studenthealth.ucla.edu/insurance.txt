    var data = await r.json();
    node.innerHTML = '';
    var toc = document.createElement('div');
    toc.classList.add('table-of-contents');
    node.parentNode.insertBefore(toc, node);
    console.log(node.parentNode.parentNode);
    var activeCategory = '';
    for(var i=0; i<data.length; i++) {
        var item = data[i];
        if(activeCategory != item.category) {
            var h2 = document.createElement('h2');
            h2.innerHTML = item.category;
            activeCategory = `${item.category}`;
            node.appendChild(h2);
        }
        var h3 = document.createElement('h3');
        h3.innerHTML = item.question;

        var div = document.createElement('div');
        div.innerHTML = item.answer;

        for(var ul of div.querySelectorAll('ul')) {

                a.setAttribute('href', `https:///www.uclahealth.org/${a.getAttribute('href')}`);
            }
        }

        node.appendChild(h3);
        node.appendChild(div);
    }

function generateTableOfContents(toc) {

    }

    for(var i=0; i < items.length; i++) {
        var item = items[i];
        console.log(item);
        var key = item.textContent.replace(/ /g, '_').replace(/[^a-zA-Z0-9_]/g, '').toLowerCase();
        item.id = key;
        var li = document.createElement('li');        
        var a = document.createElement('a');
        a.href = '#' + key;
        a.innerText = item.innerText;
        li.appendChild(a);

  <div class="body-content paragraph--section" data-section-title="The Ashe Center is UCSHIP Students&#039; Medical Home">
          <h2>
The Ashe Center is UCSHIP Students&#039; Medical Home
</h2>
      
<p>&nbsp;</p>

<section class="columns columns-2 columns-2-5050">
<section class="column column-1">
<p><strong>The Ashe Center&nbsp;</strong>serves as every UC SHIP student's&nbsp;<em>medical home</em>. Medical services must be initiated at The Ashe Center and&nbsp;a referral&nbsp;are required for all non-emergent services outside of Ashe. Claims will not be considered nor paid by Anthem Blue Cross without a valid Ashe Center referral.<br>
<br>
Before seeking medical or mental health treatment anywhere outside of the Ashe Center or CAPS (e.g. out of state/out of country)&nbsp;you must receive a referral.</p>

<p>Mental health/substance abuse services must be initiated at&nbsp;<a href="https://www.counseling.ucla.edu/contact-us">UCLA Counseling and Psychological Services (CAPS)</a>&nbsp;and continued care may be referred to UCLA Behavioral Health Services or other appropriate resources.</p>

<p class="button"><a href="https://www.studenthealth.ucla.edu/insurance/benefits/referrals">Learn About Referrals</a></p>
</section>

<section class="column column-2">
<section class="callout">
<h3>Waiving UCSHIP? Consider BruinCare!</h3>

<p>The Ashe Center wants all UCLA students to feel comfortable accessing all of their medical needs through our facility, which is why we developed BruinCare. While many students waive UCSHIP, these students are likely to still utilize Ashe services because of our center's convenience and central location. All registered students can utilize The Ashe Center. BruinCare is a program that gives students who have waived UCSHIP prepaid access to certain services at The Ashe Center. By purchasing BruinCare during the waiver process, you eliminate the $25 visit fee, and more.</p>

<p class="button"><a href="/insurance/bruincare">Learn more about BruinCare</a></p>
</section>
