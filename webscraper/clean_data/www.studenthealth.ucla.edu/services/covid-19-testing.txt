"use strict";
async function covidFAQ() {
    var node = document.querySelector('[data-ucla=covid-faq]');
    if(typeof(node) == undefined || node == null) {
        return;
    }
    var before = `${node.innerHTML}`;
    node.innerText = 'Loading FAQs...';
    var r = await fetch('https://www.uclahealth.org/ucla-campus-faq-feed.cfm?smajax=true');
    var data = await r.json();
    node.innerHTML = '';
    var toc = document.createElement('div');
    toc.classList.add('table-of-contents');
    node.parentNode.insertBefore(toc, node);
    console.log(node.parentNode.parentNode);
    var activeCategory = '';
    for(var i=0; i<data.length; i++) {
        var item = data[i];
        if(activeCategory != item.category) {
            var h2 = document.createElement('h2');
            h2.innerHTML = item.category;
            activeCategory = `${item.category}`;
            node.appendChild(h2);
        }
        var h3 = document.createElement('h3');
        h3.innerHTML = item.question;

        var div = document.createElement('div');
        div.innerHTML = item.answer;

        for(var ul of div.querySelectorAll('ul')) {
            ul.classList.add('bullet');
        }
        for(var a of div.querySelectorAll('a')) {
            if(!a.getAttribute('href').match(/http/)) {
                a.setAttribute('href', `https:///www.uclahealth.org/${a.getAttribute('href')}`);
            }
        }

        node.appendChild(h3);
        node.appendChild(div);
    }

var tocs = document.querySelectorAll('.table-of-contents');
for(var i=0; i < tocs.length; i++) {
    generateTableOfContents(tocs[i]);
}

 


function generateTableOfContents(toc) {
    var parent = document.querySelector('.page-content');
    toc.innerHTML = '';
    var currentLevel = 2;
    var ulForLevel = {
        2: document.createElement('ul'),
        3: null
    }
    toc.appendChild(ulForLevel[2]);
    console.log(ulForLevel);
    var tocFound = false;
    var items = parent.querySelectorAll('h2, h3');
    console.log(items);
    for(var i=0; i < items.length; i++) {
        var item = items[i];
        console.log(item);
        var key = item.textContent.replace(/ /g, '_').replace(/[^a-zA-Z0-9_]/g, '').toLowerCase();
        item.id = key;
        var li = document.createElement('li');        
        var a = document.createElement('a');
        a.href = '#' + key;
        a.innerText = item.innerText;
        li.appendChild(a);
        var level = parseInt(item.nodeName.replace('H', ''));
        if(currentLevel > level) {
            ulForLevel[currentLevel] = null;
        } else if(level == NaN) {
        	level = 2;
        }
        if(ulForLevel[level] != null) {
            ulForLevel[level].appendChild(li);           
        } else {
            ulForLevel[level] = document.createElement('ul');
            console.log( ulForLevel[level - 1], item, level);
            ulForLevel[level - 1].querySelector(':scope > li:last-child').appendChild(ulForLevel[level]);
            ulForLevel[level].appendChild(li);
        }
        currentLevel = level;
    }
}

}

if (document.readyState === "complete" || document.readyState === "interactive") {
    // Document is already ready to go.
    setTimeout(covidFAQ, 1);
} else {
    document.addEventListener('DOMContentLoaded', function (event) {
        covidFAQ();
    });
}
