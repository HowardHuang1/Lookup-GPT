"use strict";
async function covidFAQ() {
    var node = document.querySelector('[data-ucla=covid-faq]');
    if(typeof(node) == undefined || node == null) {
        return;
    }
    var before = `${node.innerHTML}`;
    node.innerText = 'Loading FAQs...';
    var r = await fetch('https://www.uclahealth.org/ucla-campus-faq-feed.cfm?smajax=true');
    var data = await r.json();
    node.innerHTML = '';
    var toc = document.createElement('div');
    toc.classList.add('table-of-contents');
    node.parentNode.insertBefore(toc, node);
    console.log(node.parentNode.parentNode);
    var activeCategory = '';
    for(var i=0; i<data.length; i++) {
        var item = data[i];
        if(activeCategory != item.category) {
            var h2 = document.createElement('h2');
            h2.innerHTML = item.category;
            activeCategory = `${item.category}`;
            node.appendChild(h2);
        }
        var h3 = document.createElement('h3');
        h3.innerHTML = item.question;

        var div = document.createElement('div');
        div.innerHTML = item.answer;

        for(var ul of div.querySelectorAll('ul')) {
            ul.classList.add('bullet');
        }
        for(var a of div.querySelectorAll('a')) {
            if(!a.getAttribute('href').match(/http/)) {
                a.setAttribute('href', `https:///www.uclahealth.org/${a.getAttribute('href')}`);
            }
        }

        node.appendChild(h3);
        node.appendChild(div);
    }

var tocs = document.querySelectorAll('.table-of-contents');
for(var i=0; i < tocs.length; i++) {
    generateTableOfContents(tocs[i]);
}

 


function generateTableOfContents(toc) {
    var parent = document.querySelector('.page-content');
    toc.innerHTML = '';
    var currentLevel = 2;
    var ulForLevel = {
        2: document.createElement('ul'),
        3: null
    }
    toc.appendChild(ulForLevel[2]);
    console.log(ulForLevel);
    var tocFound = false;
    var items = parent.querySelectorAll('h2, h3');
    console.log(items);
    for(var i=0; i < items.length; i++) {
        var item = items[i];
        console.log(item);
        var key = item.textContent.replace(/ /g, '_').replace(/[^a-zA-Z0-9_]/g, '').toLowerCase();
        item.id = key;
        var li = document.createElement('li');        
        var a = document.createElement('a');
        a.href = '#' + key;
        a.innerText = item.innerText;
        li.appendChild(a);
        var level = parseInt(item.nodeName.replace('H', ''));
        if(currentLevel > level) {
            ulForLevel[currentLevel] = null;
        } else if(level == NaN) {
        	level = 2;
        }
        if(ulForLevel[level] != null) {
            ulForLevel[level].appendChild(li);           
        } else {
            ulForLevel[level] = document.createElement('ul');
            console.log( ulForLevel[level - 1], item, level);
            ulForLevel[level - 1].querySelector(':scope > li:last-child').appendChild(ulForLevel[level]);
            ulForLevel[level].appendChild(li);
        }
        currentLevel = level;
    }
}

}

} else {
    document.addEventListener('DOMContentLoaded', function (event) {
        covidFAQ();
    });
}

<p class="lead">Online video appointments (telemedicine) with an Anthem network provider when The Ashe Center is closed.</p> <p>Your UC SHIP plan makes seeing a doctor easy when The Ashe Center is not available to you (building closures, away from campus, or if you just have a very busy schedule that prevents you from scheduling an in-person appointment).</p> <p>LiveHealth Online provides access to U.S. board-certified doctors 24/7/365 via phone or online video consults for urgent, non-Emergency medical assistance, including the ability to write prescriptions. LiveHealth Online provides behavioral health services as well. If follow up care is required after a consult, the student must obtain a referral from the Ashe Center prior to seeking care outside of the Ashe Center.</p> <section class="columns columns-2 columns-2-5050"> <section class="column column-1"> <h2>Scheduling</h2> <p>This service is available by registering and going to <a href="http://www.livehealthonline.com/">www.livehealthonline.com</a>, or by calling 1-888-LiveHealth.</p> <ul> <li>Either go to LiveHealth Online or download the mobile app through Google Play or the App Store</li> <li>Register using your UC SHIP Anthem ID number (without UC SHIP, just sign up without an insurance plan. Note: Waiver students pay $59)</li> <li>For the question: “Do you have health insurance?” Select “Yes.”</li> <li>From the Health Plan drop-down, select “Anthem Blue Cross (CA)”</li> <li>Enter your Subscriber ID (Your Subscriber ID is found on your UC SHIP</li> <li>insurance card — you can access this using the <a href="https://www.studenthealth.ucla.edu/services/sydney-health-mobile-app">SydneyHealth mobile app</a>.</li> <li>Your subscriber ID will begin with the characters XDP.)</li> <li>For the question: “Are you the primary subscriber?” Select “Yes.”</li> <li>Click “Finish” and you will be directed to The Ashe Center page for</li> <li>accessing LiveHealth sessions.</li> <li>Lastly, for the best coordination of your care, please select the option at the end of the visit to send post-visit records to The Ashe Center. The email where you will send these records is: <a href="mailto:%61%73%68%65%6d%65%64%69%63%61%6c%72%65%63%6f%72%64%73%40%61%73%68%65%2e%75%63%6c%61%2e%65%64%75">ashemedicalrecords@ashe.ucla.edu</a>.</li></ul></section> <section class="column column-2"> <h2>Hours</h2> <p>LiveHealth Online is available 24 hours a day on weekdays, all weekends and campus holidays.</p> <h2>Private Space Reservations</h2> <p>Need a private space to attend your telehealth visit your Ashe Center provider? Private spaces are now available by reservation thanks to our partnership with UCLA Housing and Residential Life. Here's how it works:</p> <ul> <li>Schedule your telehealth appointment online (using the Patient Portal) or by calling The Ashe Center main line (310-825-4073, Option 1).</li> <li>Our Scheduling Team can reserve a private space for you at Covel Commons on The HIll to attend your telehealth appointment.</li> <li>Questions? Call us and speak to The Ashe Center Scheduling team by phone.</li></ul></section></section> <section class="columns columns-2 columns-2-5050"> <section class="column column-1"> <h2>What to Bring</h2> <ul> <li>Locate a quiet, secure location to begin your LiveHealth Online session</li></ul></section> <section class="column column-2"> <h2>Fees</h2> <p>Patients eligible to be seen are current students, past students of the most previous academic term, and other UC campus students.</p> <h3>UCSHIP</h3> <p>Currently registered UCLA students</p> <h4>Visits</h4> <ul> <li>LiveHealth Online Visit $0</li></ul> <h3>Waived UCSHIP</h3> <p>Currently registered UCLA students who have waived UCSHIP</p> <h4>Visits</h4> <ul> <li>LiveHealth Online Visit $59</li></ul> <h3>BruinCare</h3> <p>Currently registered UCLA students who have waived UCSHIP and bought BruinCare to get unlimited access to core services during the academic year</p> <h4>Visits</h4> <ul> <li>LiveHealth Online Visit $59</li></ul> <h3>Fee for Service</h3> <p>Students from other UC campuses or UCLA students who are not currently registered, such as during summer session or who were registered in a recent term</p> <h4>Visits</h4> <ul> <li>LiveHealth Online Visit $59</li></ul></section></section>
