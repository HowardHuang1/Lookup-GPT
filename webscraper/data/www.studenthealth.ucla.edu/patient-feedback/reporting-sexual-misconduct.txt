async function covidFAQ() {
    var node = document.querySelector('[data-ucla=covid-faq]');
    if(typeof(node) == undefined || node == null) {
        return;
    }
    var before = `${node.innerHTML}`;
    node.innerText = 'Loading FAQs...';
    var r = await fetch('https://www.uclahealth.org/ucla-campus-faq-feed.cfm?smajax=true');
    var data = await r.json();
    node.innerHTML = '';
    var toc = document.createElement('div');
    toc.classList.add('table-of-contents');
    node.parentNode.insertBefore(toc, node);
    console.log(node.parentNode.parentNode);
    var activeCategory = '';
    for(var i=0; i<data.length; i++) {
        var item = data[i];
        if(activeCategory != item.category) {
            var h2 = document.createElement('h2');
            h2.innerHTML = item.category;
            activeCategory = `${item.category}`;
            node.appendChild(h2);
        }
        var h3 = document.createElement('h3');
        h3.innerHTML = item.question;

        var div = document.createElement('div');
        div.innerHTML = item.answer;

        for(var ul of div.querySelectorAll('ul')) {
            ul.classList.add('bullet');
        }
        for(var a of div.querySelectorAll('a')) {
            if(!a.getAttribute('href').match(/http/)) {
                a.setAttribute('href', `https:///www.uclahealth.org/${a.getAttribute('href')}`);
            }
        }

        node.appendChild(h3);
        node.appendChild(div);
    }

var tocs = document.querySelectorAll('.table-of-contents');
for(var i=0; i < tocs.length; i++) {
    generateTableOfContents(tocs[i]);
}

 


function generateTableOfContents(toc) {
    var parent = document.querySelector('.page-content');
    toc.innerHTML = '';
    var currentLevel = 2;
    var ulForLevel = {
        2: document.createElement('ul'),
        3: null
    }
    toc.appendChild(ulForLevel[2]);
    console.log(ulForLevel);
    var tocFound = false;
    var items = parent.querySelectorAll('h2, h3');
    console.log(items);
    for(var i=0; i < items.length; i++) {
        var item = items[i];
        console.log(item);
        var key = item.textContent.replace(/ /g, '_').replace(/[^a-zA-Z0-9_]/g, '').toLowerCase();
        item.id = key;
        var li = document.createElement('li');        
        var a = document.createElement('a');
        a.href = '#' + key;
        a.innerText = item.innerText;
        li.appendChild(a);
        var level = parseInt(item.nodeName.replace('H', ''));
        if(currentLevel > level) {
            ulForLevel[currentLevel] = null;
        } else if(level == NaN) {
        	level = 2;
        }
        if(ulForLevel[level] != null) {
            ulForLevel[level].appendChild(li);           
        } else {
            ulForLevel[level] = document.createElement('ul');
            console.log( ulForLevel[level - 1], item, level);
            ulForLevel[level - 1].querySelector(':scope > li:last-child').appendChild(ulForLevel[level]);
            ulForLevel[level].appendChild(li);
        }
        currentLevel = level;
    }
}

<p class="lead">If you have experienced sexual harassment, please report it. To speak with a Title IX officer, please call (310) 206-3417 .</p> <h2>Reporting Sexual Misconduct</h2> <h3>How to Make a Report of Sexual Harassment and Notice of Nondiscrimination</h3> <p>The University of California and UCLA Arthur Ashe Student Health and Wellness Center are committed to the principle of treating each community member with respect and dignity.</p> <h3>University of California Notice of Nondiscrimination</h3> <p>Across all of its locations, including clinical settings, the University prohibits discrimination and harassment (including sexual assault and other forms of sexual harassment). UCLA Arthur Ashe Student Health and Wellness Center provides equal opportunities for all patients, applicants, employees, faculty, and students regardless of sex, gender identity or expression, sexual orientation, or other legally protected characteristics. For more on the University’s commitment to integrity, accountability and respect for all members of our community, please see the University’s Statement of Ethical Values and Standards of Ethical Conduct at <a href="https://www.compliance.ucla.edu/statement-of-ethical-values">https://www.compliance.ucla.edu/statement-of-ethical-values</a>.</p> <h3>UCLA Patients and Students - How to Report</h3> <p>If you have experienced sexual harassment, please report it. To speak with a Title IX officer, please call <a href="tel:3102063417">(310) 206-3417</a>. You may also <a href="https://ucla-gme-advocate.symplicity.com/public_report/index.php/pid215484?">submit an online report</a>.</p> <p>The UCLA Title IX Office tries to protect people’s privacy, while also fulfilling its obligations to respond appropriately to all reports it receives. In some situations, certain laws and University policies may require disclosure of reports to others, such as other University officials, licensing boards, law enforcement, or other agencies. If you are unsure if you want to file a report, we recommend contacting one of the following confidential resources first for advice and support:</p> <ul> <li>CARE Advocate- Advocacy Office for Sexual and Gender Based Violence and Misconduct</li> <li>CARE Advocates are available to support and advocate for UCLA student survivors of sexual violence or sexual harassment. They can assist in finding resources and navigating reporting options. You can get help from a CARE Advocate without formally reporting sexual violence.</li> <li>Murphy A Level, Room 233 | <a href="tel:3102062465">310-206-2465</a> | <a href="mailto:CAREadvocate@careprogram.ucla.edu">CAREadvocate@careprogram.ucla.edu</a></li> <li><a href="https://www.careprogram.ucla.edu/">https://www.careprogram.ucla.edu/</a></li> <li>CAPS (Counseling and Psychological Services)</li> <li>For emotional support, or individual or group counseling.</li> <li>John Wooden Center West | <a href="tel:310-825-0768">310-825-0768</a> (24/7) | <a href="https://www.counseling.ucla.edu/">https://www.counseling.ucla.edu/</a></li> <li>Office Hours: Mon-Thurs, 8am-8pm; Fri 9am-5</li> <li>Rape Treatment Center, Santa Monica</li> <li>Free medical treatment, preventive health care and counseling available 24/7.</li> <li>A sexual assault survivor has the option of a Sexual Assault Forensic Exam (SAFE) (commonly referred to as “rape exam”) without filing a police report. This is referred to as a Non-Investigative Report (NIR). UCPD will provide free and confidential transportation to the RTC.</li> <li>1250 Sixteenth Street, Santa Monica CA 90404 | <a href="tel:4242596000">424-259-6000</a> | <a href="http://www.rapetreatmentcenter.org">www.rapetreatmentcenter.org</a></li> <li>Student Legal Services</li> <li>Confidential legal counseling and assistance</li> <li>A239 Murphy Hall | <a href="tel:3108259894">310-825-9894</a> | <a href="mailto:slegal@saonet.ucla.edu">slegal@saonet.ucla.edu</a> | <a href="https://www.studentlegal.ucla.edu/">www.studentlegal.ucla.edu</a></li></ul> <p>These offices are not required to report instances of sexual harassment to the University.</p> <p>The University reviews all reports of sexual harassment in accordance with the <a href="https://policy.ucop.edu/doc/4000385/SVSH">University’s Sexual Violence and Sexual Harassment Policy</a> and other applicable policies.</p> <h3>UCLA Workforce - How to Report</h3> <p>UCLA provides <a href="https://sexualharassment.ucla.edu/resources/resources-for-complainants">resources to faculty and staff</a> who want to report sexual misconduct or seek help for themselves or others.</p> <p>As an institution, we are all responsible for creating and maintaining the safe, supportive and respectful environment our patients and colleagues expect and deserve. All members of the UCLA community are urged to immediately report information regarding possible sexual misconduct directly to the Title IX office or anonymously to the compliance hotline. More information about <a href="https://sexualviolence.universityofcalifornia.edu/faq/#1_3">your responsibility to report sexual misconduct as a member of the UC community</a>.</p> <h4>Title IX:</h4> <ul> <li>Phone: <a href="tel:3102063417">(310) 206-3417</a></li> <li>Email: <a href="mailto:TitleIX@conet.ucla.edu">TitleIX@conet.ucla.edu</a></li> <li><a href="https://ucla-gme-advocate.symplicity.com/public_report/index.php/pid215484?">Public Incident Report Form</a> (you may submit an anonymous report through this form)</li></ul> <h4>Compliance Hotline:</h4> <p><a href="tel:8004034744">(800) 403-4744</a></p> <h3>To Provide General Feedback about your Patient Care Experience</h3> <p>UCLA Arthur Ashe Student Health and Wellness Center is interested in hearing about your overall care experience. To provide general feedback about your visit or experience with a UCLA Arthur Ashe Student Health and Wellness Center provider, or to communicate any complaint about unprofessional conduct, please <a href="https://www.studenthealth.ucla.edu/patient-feedback">submit feedback here</a>.</p> <h3>To Obtain Your Medical Records</h3> <p>You may obtain a copy of your medical records in person or by mail. For detailed instructions, visit the UCLA Arthur Ashe Student Health and Wellness Center <a href="/contact/medical-records">Medical Records page</a>.</p> <h3>For Hearing and Visual Access Assistance</h3> <p>We provide free aids and services to people with disabilities to communicate effectively with us, like qualified sign language interpreters and written information in other formats (large print, audio, accessible electronic formats, other formats). If you need these services, contact <a href="https://www.cae.ucla.edu/">UCLA Center for Accessible Education</a>, <a href="tel:3108251501">(310) 825-1501</a>.</p> <h3>Reporting Obligations of UC Personnel</h3> <p>All University employees are responsible for creating and maintaining the safe, supportive and respectful environment that our patients and colleagues expect and deserve. Additionally, under some circumstances, UCLA Arthur Ashe Student Health and Wellness Center staff members are required by University policy to immediately report possible sexual harassment to the Title IX office. Even those who are not subject to a mandate are urged to immediately report information regarding possible sexual harassment directly to the Title IX office or anonymously to the compliance hotline. More information about your <a href="https://www.sexualviolence.ucla.edu/FAQS/Responsible-Employees">responsibility to report sexual harassment as a member of the UC community</a>.</p>
